344template <typename T>
345bool IChannel::GetChannelValue(const std::vector<uint8_t> &record_buffer,
346 T &dest) const {
347 bool valid = false;
348 switch (DataType()) {
349 case ChannelDataType::UnsignedIntegerLe:
350 case ChannelDataType::UnsignedIntegerBe: {
351 uint64_t value = 0;
352 valid = GetUnsignedValue(record_buffer, value);
353 dest = static_cast<T>(value);
354 break;
355 }
356
357 case ChannelDataType::SignedIntegerLe:
358 case ChannelDataType::SignedIntegerBe: {
359 int64_t value = 0;
360 valid = GetSignedValue(record_buffer, value);
361 dest = static_cast<T>(value);
362 break;
363 }
364 case ChannelDataType::FloatLe:
365 case ChannelDataType::FloatBe: {
366 double value = 0;
367 valid = GetFloatValue(record_buffer, value);
368 dest = static_cast<T>(value);
369 break;
370 }
371
372 case ChannelDataType::StringUTF16Le:
373 case ChannelDataType::StringUTF16Be:
374 case ChannelDataType::StringUTF8:
375 case ChannelDataType::StringAscii: {
376 std::string text;
377 valid = GetTextValue(record_buffer, text);
378 std::istringstream data(text);
379 data >> dest;
380 break;
381 }
382
383 case ChannelDataType::MimeStream:
384 case ChannelDataType::MimeSample:
385 case ChannelDataType::ByteArray: {
386 std::vector<uint8_t> list;
387 valid = GetByteArrayValue(record_buffer, list);
388 dest = list.empty() ? T{} : list[0];
389 break;
390 }
391
392 case ChannelDataType::CanOpenDate: {
393 uint64_t ms_since_1970 = 0;
394 valid = GetCanOpenDate(record_buffer, ms_since_1970);
395 dest = static_cast<T>(ms_since_1970);
396 break;
397 }
398
399 case ChannelDataType::CanOpenTime: {
400 uint64_t ms_since_1970 = 0;
401 valid = GetCanOpenTime(record_buffer, ms_since_1970);
402 dest = static_cast<T>(ms_since_1970);
403 break;
404 }
405 default:
406 break;
407 }
408 return valid;
409}

421template <typename T>
422void IChannel::SetChannelValue(const T &value, bool valid) {
423 switch (DataType()) {
424 case ChannelDataType::UnsignedIntegerLe:
425 SetUnsignedValueLe(static_cast<uint64_t>(value), valid);
426 break;
427
428 case ChannelDataType::UnsignedIntegerBe:
429 SetUnsignedValueBe(static_cast<uint64_t>(value), valid);
430 break;
431
432 case ChannelDataType::SignedIntegerLe:
433 SetSignedValueLe(static_cast<int64_t>(value), valid);
434 break;
435
436 case ChannelDataType::SignedIntegerBe:
437 SetSignedValueBe(static_cast<int64_t>(value), valid);
438 break;
439
440 case ChannelDataType::FloatLe:
441 SetFloatValueLe(static_cast<double>(value), valid);
442 break;
443
444 case ChannelDataType::FloatBe:
445 SetFloatValueBe(static_cast<double>(value), valid);
446 break;
447
448 case ChannelDataType::StringUTF8:
449 case ChannelDataType::StringAscii:
450 case ChannelDataType::StringUTF16Be:
451 case ChannelDataType::StringUTF16Le:
452 SetTextValue(std::to_string(value), valid);
453 break;
454
455 case ChannelDataType::MimeStream:
456 case ChannelDataType::MimeSample:
457 case ChannelDataType::ByteArray:
458 // SetByteArray(value, valid);
459 break;
460
461 case ChannelDataType::CanOpenDate:
462 if (typeid(T) == typeid(uint64_t) && DataBytes() == 7) {
463 const auto date_array =
464 MdfHelper::NsToCanOpenDateArray(static_cast<uint64_t>(value));
465 SetByteArray(date_array, valid);
466 } else {
467 SetValid(false);
468 }
469 break;
470
471 case ChannelDataType::CanOpenTime:
472 if (typeid(T) == typeid(uint64_t) && DataBytes() == 6) {
473 const auto time_array =
474 MdfHelper::NsToCanOpenTimeArray(static_cast<uint64_t>(value));
475 SetByteArray(time_array, valid);
476 } else {
477 SetValid(false);
478 }
479 break;
480
481
482 default:
483 SetValid(false);
484 break;
485 }

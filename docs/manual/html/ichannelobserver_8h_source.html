85bool IChannelObserver::GetChannelValue(uint64_t sample, V& value) const {
86 bool valid = false;
87 value = {};
88 switch (channel_.DataType()) {
89 case ChannelDataType::CanOpenDate:
90 case ChannelDataType::CanOpenTime: {
91 uint64_t v = 0; // ns since 1970
92 valid = GetSampleUnsigned(sample, v);
93 value = static_cast<V>(v);
94 break;
95 }
96
97 case ChannelDataType::UnsignedIntegerLe:
98 case ChannelDataType::UnsignedIntegerBe: {
99 uint64_t v = 0;
100 valid = GetSampleUnsigned(sample, v);
101 value = static_cast<V>(v);
102 break;
103 }
104
105 case ChannelDataType::SignedIntegerLe:
106 case ChannelDataType::SignedIntegerBe: {
107 int64_t v = 0;
108 valid = GetSampleSigned(sample, v);
109 value = static_cast<V>(v);
110 break;
111 }
112
113 case ChannelDataType::FloatLe:
114 case ChannelDataType::FloatBe: {
115 double v = 0.0;
116 valid = GetSampleFloat(sample, v);
117 value = static_cast<V>(v);
118 break;
119 }
120
121 case ChannelDataType::StringAscii:
122 case ChannelDataType::StringUTF8:
123 case ChannelDataType::StringUTF16Le:
124 case ChannelDataType::StringUTF16Be: {
125 std::string v;
126 valid = GetSampleText(sample, v);
127 std::istringstream s(v);
128 s >> value;
129 break;
130 }
131
132 case ChannelDataType::MimeStream:
133 case ChannelDataType::MimeSample:
134 case ChannelDataType::ByteArray: {
135 std::vector<uint8_t> v;
136 valid = GetSampleByteArray(sample, v);
137 value = static_cast<V>(v.empty() ? V{} : v[0]);
138 break;
139 }
140
141 default:
142 break;
143 }
144 return valid;
145}

158template <typename V>
159bool IChannelObserver::GetEngValue(uint64_t sample, V& value) const {
160 const auto* conversion = channel_.ChannelConversion();
161 if (conversion == nullptr) {
162 return GetChannelValue(sample, value);
163 }
164
165 bool valid = false;
166 value = {};
167 switch (channel_.DataType()) {
168 case ChannelDataType::UnsignedIntegerLe:
169 case ChannelDataType::UnsignedIntegerBe: {
170 uint64_t v = 0;
171 valid = GetSampleUnsigned(sample, v) && conversion->Convert(v, value);
172 break;
173 }
174
175 case ChannelDataType::SignedIntegerLe:
176 case ChannelDataType::SignedIntegerBe: {
177 int64_t v = 0;
178 valid = GetSampleSigned(sample, v) && conversion->Convert(v, value);
179 break;
180 }
181
182 case ChannelDataType::FloatLe:
183 case ChannelDataType::FloatBe: {
184 double v = 0.0;
185 valid = GetSampleFloat(sample, v) && conversion->Convert(v, value);
186 break;
187 }
188
189 case ChannelDataType::CanOpenDate:
190 case ChannelDataType::CanOpenTime:
191 valid = GetChannelValue(sample, value); // No conversion is allowed;
192 break;
193
194 default:
195 break;
196 }
197 return valid;
198}
